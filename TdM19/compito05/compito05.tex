\begin{exercise}
Let $a=\<a_i:i\in I\>$ be an (infinite) $A$-indiscernible sequence and let $I\subseteq J$ with $|J|\le \kappa$. Then there is an $A$-indiscernible sequence $c=\<c_i:i\in J\>$ such that $c_{\restriction I}=a$.
\end{exercise}

\begin{exercise} 
Let $a\equiv_A b$ and let $q(x)=\tp(a/A)=\tp(b/A)$. Prove that if $q(x)$ extends to a global $A$-invariant type $p(x)\in S(\U)$ then there is a sequence $c=\<c_i:i<\omega\>$ such that $a,c$ and $b,c$ are both sequences of $A$-indiscernibles.
\end{exercise}

\begin{exercise} 
Prove that the following are equivalent for every $\D$ that is externally definable by a stable formula
\begin{itemize}
\item[1.] $\D$ is invariant over every model $M$ containing $A$;
\item[2.] $\D\in\acl^\eq\!A$.
\end{itemize}
\end{exercise}
