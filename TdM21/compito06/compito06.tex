\begin{exercise} 
Let $\phi(x\,;y)\in L(\U)$. Prove that the following are equivalent
  \begin{itemize}
  \item[1.] there is a sequence $\<a_i\,:\,i\in\omega\>$ such that 
  $\phi(\U\,;a_i)\subset\phi(\U\,;a_{i+1})$ for every $i<\omega$;

  \item[2.] there is a sequence $\<a_i\,:\,i\in\omega\>$ such that 
  $\phi(\U\,;a_{i+1})\subset\phi(\U\,;a_i)$ for every $i<\omega$. 
  \end{itemize}
\end{exercise}

\begin{exercise}  
  Let $\phi(x,y)\in L(\U)$. 
  Prove that if the set $\big\{\phi(a,\U)\ :\ a\in\U^{|x|}\big\}$ is infinite then it has cardinality $\kappa$.
  Does the claim remain true with a type $p(x,y)\subseteq L(A)$ for $\phi(x,y)$?

  Suggerimento: si cerchi un controesempio con $\U\equiv\NN$ nel linguaggio degli ordini.
\end{exercise}
