\begin{exercise}
Let $\phi(z)\in L(A)$ be a consistent formula.
Prove that, if $a\in\acl(A,b)$ for every $b\models\phi(z)$, then $a\in\acl(A)$.
Prove the same claim with a type $p(z)\subseteq L(A)$ for $\phi(z)$.
\end{exercise}

\begin{exercise}
Let $a\in\U\sm\acl\0$.
Prove that $\U$ is isomorphic to some elementary substructure $\V\preceq \U$ such that $a\notin\V$.
\end{exercise}

\begin{exercise}
Let $C$ be a finite set.
Prove that if $C\cap M\neq\0$ for every model $M$ containing $A$, then $C\cap\acl(A)\neq\0$.
\end{exercise}
