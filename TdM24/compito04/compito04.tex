\begin{exercise}
Let $\varphi(z)\in L(A)$ be a consistent formula.
Let $p(z)\subseteq L(A)$ be a consistent type.
Prove that, if $a\in\acl(A,b)$ for every $b\models p(z)$, then $a\in\acl(A)$.
\end{exercise}

\begin{exercise}
  Let $a\in\U\sm\acl\0$.
  Prove that $\U$ is isomorphic to some $\V\preceq \U$ such that $a\notin\V$.
\end{exercise}

\begin{exercise}
  Let $C$ be a finite set.
  Prove that if $C\cap M\neq\0$ for every model $M$ containing $A$, then $C\cap\acl(A)\neq\0$.
\end{exercise}
  
\begin{exercise}
  Prove that for every $A\subseteq N$ there is an $M$ such that $\acl A\,=\,M\cap N$.
\end{exercise}
