\begin{exercise}
  Prove that the following are equivalent
  \begin{itemize}
  \item[1.] $T$ has weak elimination of imaginaries;
  \item[2.] for ${\EuScript A}\in \U^\eq$, there is a smallest set $A{=}\acl A\subseteq\U$ such that ${\EuScript A}\in\dcl^\eq A$;
  \item[3.] for $a,b\in\U$, if ${\EuScript A}\in\dcl^\eq(a)\cap\dcl^\eq(b)$ then ${\EuScript A}\in\dcl^\eq\big(\acl a\cap\acl b\big)$.
  \end{itemize} 
\end{exercise}
